 void LoadTasks(){
        ifstream file("tasks.txt");
        
        if (file.is_open())
        {
            m_vTasks.clear();
            bool completed;
            string taskDescription;

            while(file >> completed){
                getline(file >> ws,taskDescription);
                auto ptr = make_unique<Task>(taskDescription);
                if (completed)
                {
                    ptr->CompletedTask();
                }
                m_vTasks.push_back(move(ptr));
            }
        }
    }
